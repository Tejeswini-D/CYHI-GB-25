  <div class="card">
    <h1>GB Wallet</h1>

    <label>Bet Amount</label>
    <input type="number" id="betAmount" value="10"><br>

    <a href="timer.html">
    <button onclick="placeBet()">Place Bet</button></a>
    <button onclick="finishTask(true)">Completed the task</button>
    <button onclick="finishTask(false)">Give up on the task</button>
  </div>

  <script>
    let balance = 100;
    let betAmount = 0;

    function updateBalance() {
      document.getElementById("balance").textContent = balance;
    }

    function placeBet() {
      const amount = parseInt(document.getElementById("betAmount").value);
      if (amount > balance || amount <= 0) {
        alert("Invalid bet amount!");
        return;
      }
      betAmount = amount;
      balance -= amount;
      updateBalance();
      alert(You placed a bet of ${amount} GB);
    }

    function finishTask(success) {
      if (betAmount === 0) {
        alert("No active bet!");
        return;
      }
      if (success) {
        balance += betAmount * 2;
        alert(Completed the task! Gained ${betAmount * 2} GB);
      } else {
        alert(Gave up on the task. Lost ${betAmount} GB);
      }
      betAmount = 0;
      updateBalance();
    }
  </script>
